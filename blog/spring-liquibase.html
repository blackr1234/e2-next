5 Unit test
我地可以用 JUnit 5 測試 Liquibase 既 migration 以及 rollback 功能。
AbstractLiquibaseTest.java：
1@Slf4j
2@SpringBootTest(
3    classes = {
4        TestApp.class,
5    }
6)
7@DirtiesContext(classMode = DirtiesContext.ClassMode.AFTER_EACH_TEST_METHOD)
8public abstract class AbstractLiquibaseTest {
9
10    @Autowired DataSource dataSource;
11    @Autowired ResourceLoader resourceLoader;
12
13    @SpringBootApplication
14    public static class TestApp {}
15
16    private static record Context(Liquibase liquibase, Connection connection) {}
17
18    @FunctionalInterface
19    public static interface ThrowingRunnable {
20        void run() throws Exception;
21    }
22
23
24
25
26    private Context prepareContext() throws Exception {
27
28        try {
29            final Connection connection = dataSource.getConnection();
30            System.out.println(connection);
31
32            final DatabaseMetaData metadata = connection.getMetaData();
33            log.info("Data source URL: {}", metadata.getURL());
34            log.info("Data source username: {}", metadata.getUserName());
35
36            final Liquibase liquibase = new Liquibase(
37                    "db/changelog/db.changelog-master.yaml",
38                    new SpringResourceAccessor(resourceLoader),
39                    DatabaseFactory.getInstance().findCorrectDatabaseImplementation(new JdbcConnection(connection)));
40
41            log.info("Initialized Liquibase.");
42
43            return new Context(liquibase, connection);
44
45        } catch (Exception e) {
46            log.error("Failed to initialize Liquibase.", e);
47            throw e;
48        }
49    }
50
51    private void close(Context context) {
52
53        try {
54            context.liquibase().close();
55            log.info("Closed Liquibase.");
56        } catch (Exception e) {
57            log.error("Failed to close Liquibase.", e);
58        }
59
60        try {
61            context.connection().close();
62            log.info("Closed database connection.");
63        } catch (Exception e) {
64            log.error("Failed to close database connection.", e);
65        }
66    }
67
68    private void runWithThread(ThrowingRunnable runnable) throws Exception {
69
70        final Callable<Void> callable = () -> {
71            runnable.run();
72            return null;
73        };
74        final FutureTask<?> task = new FutureTask<>(callable);
75        final Thread thread = new Thread(task);
76        thread.start();
77        thread.join();
78
79        task.get();
80    }
81
82
83
84    protected void apply() throws Exception {
85
86        final Context context = prepareContext();
87
88        final FutureTask<?> task = new FutureTask<>(() -> {
89
90            return null;
91        });
92        final Thread thread = new Thread(task);
93        thread.start();
94        thread.join();
95
96        try {
97            runWithThread(() -> context.liquibase().update());
98            log.info("Applied Liquibase migration scripts.");
99        } catch (Exception e) {
100            log.error("Failed to perform Liquibase update.", e);
101            throw e;
102        }
103
104        try {
105            task.get();
106            log.info("Applied Liquibase migration scripts.");
107        } catch (Exception e) {
108            log.error("Failed to perform Liquibase update.", e);
109            throw e;
110        }
111
112        close(context);
113    }
114
115    protected void rollback(String tag) throws Exception {
116
117        try {
118            final Context context = prepareContext();
119
120            runWithThread(() -> context.liquibase().rollback(tag, (String) null));
121            log.info("Performed Liquibase rollback by tag = [{}].", tag);
122
123            close(context);
124
125        } catch (Exception e) {
126            log.error("Failed to perform Liquibase rollback by tag = [{}].", tag, e);
127            throw e;
128        }
129    }
130
131    protected void rollback(int count) throws Exception {
132
133        try {
134            final Context context = prepareContext();
135
136            runWithThread(() -> context.liquibase().rollback(count, null));
137            log.info("Performed Liquibase rollback by count = [{}].", count);
138
139            close(context);
140
141        } catch (Exception e) {
142            log.error("Failed to perform Liquibase rollback by count = [{}].", count, e);
143            throw e;
144        }
145    }
146}
