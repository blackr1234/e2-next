public class MainDownloader {
    private static final String OUTPUT_DIR = "C:/Users/Michael/Downloads/output_java";

    public static void main(String[] args) throws Exception {

        System.setProperty("java.util.concurrent.ForkJoinPool.common.parallelism",
                           String.valueOf(100));

        new File(OUTPUT_DIR).mkdirs();

        Jsoup.connect("https://www.dre.vanderbilt.edu/~schmidt/cs253/2023-PDFs/")
             .get()
             .select("table tbody tr > td:nth-of-type(2) a:not([href^='/'])")
             .stream()
             .parallel()
             .forEach(e -> {
                 final String fileName = e.ownText();
                 final String url = e.absUrl("href");
                 final String command = format("wget -O \"%s/%s\" \"%s\"",
                                               OUTPUT_DIR, fileName, url);

                 log.info("Downloading [{}] from: {}", fileName, url);

                 try {
                     new ProcessBuilder("cmd", "/c", command)
                         .inheritIO()
                         .start()
                         .waitFor();
                 } catch (Exception ex) {
                     log.error("Failed to execute command: {}", command, ex);
                 }
             });
    }
