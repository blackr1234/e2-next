5.2 寫 Java code

先建立一個典型既 Spring Boot web application。

OrderController.java：

@RestController
@RequestMapping(path = "orders", produces = MediaType.APPLICATION_JSON_VALUE)
public class OrderController {

    @Autowired
    RestTemplate orderRestTemplate;

    @GetMapping("{orderId}")
    public Order getOrderById(@PathVariable("orderId") String orderId) {

        // call Order Service
        return orderRestTemplate
                .exchange("http://localhost:8080/orders?orderId=" + orderId,
                          HttpMethod.GET, HttpEntity.EMPTY, Order.class)
                .getBody();
    }
}

ApiClientConfig.java：

@Configuration
public class ApiClientConfig {

    @Bean
    public RestTemplate orderRestTemplate(RestTemplateBuilder restTemplateBuilder) {
        return restTemplateBuilder.basicAuthentication("user", "pass").build();
    }
}

import static com.github.tomakehurst.wiremock.client.WireMock.aResponse;
import static com.github.tomakehurst.wiremock.client.WireMock.equalTo;
import static com.github.tomakehurst.wiremock.client.WireMock.getRequestedFor;
import static com.github.tomakehurst.wiremock.client.WireMock.stubFor;
import static com.github.tomakehurst.wiremock.client.WireMock.urlEqualTo;
import static com.github.tomakehurst.wiremock.client.WireMock.verify;
import static org.springframework.test.web.servlet.request.MockMvcRequestBuilders.get;
import static org.springframework.test.web.servlet.result.MockMvcResultHandlers.print;
import static org.springframework.test.web.servlet.result.MockMvcResultMatchers.content;
import static org.springframework.test.web.servlet.result.MockMvcResultMatchers.jsonPath;
import static org.springframework.test.web.servlet.result.MockMvcResultMatchers.status;

@AutoConfigureMockMvc
@SpringBootTest
@RunWith(SpringRunner.class)
public class OrderControllerIT {

    @Autowired
    MockMvc mockMvc;

    static WireMockServer orderServiceMockServer = new WireMockServer(8080);

    @BeforeClass
    public static void setUpOnce() {
        orderServiceMockServer.start();
    }

    @AfterClass
    public static void tearDownOnce() {
        orderServiceMockServer.stop();
    }

    @Before
    public void setUp() {
        orderServiceMockServer.resetRequests();
    }

    @Test
    public void test() throws Exception {

        stubFor(WireMock.get(urlEqualTo("/orders?orderId=1"))
                .withBasicAuth("user", "pass")
                .withQueryParam("orderId", equalTo("1"))
                .willReturn(aResponse().withHeader(HttpHeaders.CONTENT_TYPE, MediaType.APPLICATION_JSON_VALUE)
                                .withBody("{ \"orderId\": \"1\", \"totalPrice\": 8888 }")));

        mockMvc.perform(get("/orders/1"))
               .andDo(print())
               .andExpectAll(
                   status().isOk(),
                   content().json("{ \"orderId\": \"1\" }", false),
                   jsonPath("$.orderId").value("1"),
                   jsonPath("$.totalPrice").value(8888));

        verify(1, getRequestedFor(urlEqualTo("/orders?orderId=1")));
    }
}
